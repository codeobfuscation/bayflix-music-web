# /etc/nginx/sites-enabled/dev2.cwi-group.org
# Bayflix Music dev — static SPA, fronted by Cloudflare (origin IP hidden)

server {
    listen 80;
    listen [::]:80;
    server_name dev2.cwi-group.org;
    server_tokens off;

    location / {
        return 301 https://$host$request_uri;
    }
}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name dev2.cwi-group.org;
    server_tokens off;

    ssl_certificate     /etc/ssl/cloudflare/cwi-group.org.pem;
    ssl_certificate_key /etc/ssl/cloudflare/cwi-group.org.key;
    include snippets/cwi-aop.conf;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 1d;
    ssl_session_tickets off;

    # DNS resolver for proxy_pass to external hosts (artistgrid CORP-bypass).
    resolver 1.1.1.1 1.0.0.1 valid=300s ipv6=off;
    resolver_timeout 5s;

    real_ip_header CF-Connecting-IP;

    access_log /var/log/nginx/dev2.cwi-group.org.access.log;
    error_log  /var/log/nginx/dev2.cwi-group.org.error.log;

    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    root /srv/bayflix-music/web;
    index index.html;

    # CORP bypass for ArtistGrid assets — they send
    # `Cross-Origin-Resource-Policy: same-site` which Chrome enforces, blocking
    # `<img src>` and `<audio>` loads from any cross-site origin. Proxying
    # through our nginx makes them same-origin to the browser; we strip the
    # offending header and re-emit a permissive CORP.
    location ^~ /proxy/artistgrid/ {
        proxy_pass https://assets.artistgrid.cx/;
        proxy_set_header Host assets.artistgrid.cx;
        proxy_set_header User-Agent "bayflix-music-proxy";
        proxy_ssl_server_name on;
        proxy_ssl_verify off;
        proxy_hide_header Cross-Origin-Resource-Policy;
        proxy_hide_header Access-Control-Allow-Origin;
        add_header Cross-Origin-Resource-Policy "cross-origin" always;
        add_header Access-Control-Allow-Origin "*" always;
        add_header Cache-Control "public, max-age=604800" always;
        proxy_intercept_errors on;
        proxy_read_timeout 30s;
    }

    # Same trick for r2.artistgrid.cx (audio/track storage).
    location ^~ /proxy/artistgrid-r2/ {
        proxy_pass https://r2.artistgrid.cx/;
        proxy_set_header Host r2.artistgrid.cx;
        proxy_set_header User-Agent "bayflix-music-proxy";
        proxy_ssl_server_name on;
        proxy_ssl_verify off;
        proxy_hide_header Cross-Origin-Resource-Policy;
        proxy_hide_header Access-Control-Allow-Origin;
        add_header Cross-Origin-Resource-Policy "cross-origin" always;
        add_header Access-Control-Allow-Origin "*" always;
        add_header Cache-Control "public, max-age=604800" always;
        proxy_intercept_errors on;
        proxy_read_timeout 30s;
    }

    # trackerapi-{1,2,3}.artistgrid.cx — the per-sheet metadata endpoint that
    # populates /unreleased/<sheetId>. Upstream sends no CORS headers, so
    # fetch() from dev2 fails outright. Proxy here strips CORP and emits CORS.
    location ^~ /proxy/trackerapi-1/ {
        proxy_pass https://trackerapi-1.artistgrid.cx/;
        proxy_set_header Host trackerapi-1.artistgrid.cx;
        proxy_set_header User-Agent "bayflix-music-proxy";
        proxy_ssl_server_name on;
        proxy_ssl_verify off;
        proxy_hide_header Cross-Origin-Resource-Policy;
        proxy_hide_header Access-Control-Allow-Origin;
        add_header Cross-Origin-Resource-Policy "cross-origin" always;
        add_header Access-Control-Allow-Origin "*" always;
        add_header Cache-Control "public, max-age=300" always;
        proxy_intercept_errors on;
        proxy_read_timeout 30s;
    }

    location ^~ /proxy/trackerapi-2/ {
        proxy_pass https://trackerapi-2.artistgrid.cx/;
        proxy_set_header Host trackerapi-2.artistgrid.cx;
        proxy_set_header User-Agent "bayflix-music-proxy";
        proxy_ssl_server_name on;
        proxy_ssl_verify off;
        proxy_hide_header Cross-Origin-Resource-Policy;
        proxy_hide_header Access-Control-Allow-Origin;
        add_header Cross-Origin-Resource-Policy "cross-origin" always;
        add_header Access-Control-Allow-Origin "*" always;
        add_header Cache-Control "public, max-age=300" always;
        proxy_intercept_errors on;
        proxy_read_timeout 30s;
    }

    location ^~ /proxy/trackerapi-3/ {
        proxy_pass https://trackerapi-3.artistgrid.cx/;
        proxy_set_header Host trackerapi-3.artistgrid.cx;
        proxy_set_header User-Agent "bayflix-music-proxy";
        proxy_ssl_server_name on;
        proxy_ssl_verify off;
        proxy_hide_header Cross-Origin-Resource-Policy;
        proxy_hide_header Access-Control-Allow-Origin;
        add_header Cross-Origin-Resource-Policy "cross-origin" always;
        add_header Access-Control-Allow-Origin "*" always;
        add_header Cache-Control "public, max-age=300" always;
        proxy_intercept_errors on;
        proxy_read_timeout 30s;
    }

    # Qobuz API (qdl-api.monochrome.tf) — returns valid data but sends no CORS
    # headers, so browser fetch() from dev2 origin is rejected. Proxy strips
    # whatever's there and emits permissive CORS.
    location ^~ /proxy/qobuz-monochrome/ {
        proxy_pass https://qdl-api.monochrome.tf/;
        proxy_set_header Host qdl-api.monochrome.tf;
        proxy_set_header User-Agent "bayflix-music-proxy";
        proxy_ssl_server_name on;
        proxy_ssl_verify off;
        proxy_hide_header Cross-Origin-Resource-Policy;
        proxy_hide_header Access-Control-Allow-Origin;
        add_header Cross-Origin-Resource-Policy "cross-origin" always;
        add_header Access-Control-Allow-Origin "*" always;
        add_header Cache-Control "public, max-age=300" always;
        proxy_intercept_errors on;
        proxy_read_timeout 30s;
    }

    # TIDAL image CDN — `resources.tidal.com` sends no CORS so fetch() reads
    # fail. Used for cover-art blob extraction (download metadata embedding).
    location ^~ /proxy/tidal-images/ {
        proxy_pass https://resources.tidal.com/;
        proxy_set_header Host resources.tidal.com;
        proxy_set_header User-Agent "bayflix-music-proxy";
        proxy_ssl_server_name on;
        proxy_ssl_verify off;
        proxy_hide_header Cross-Origin-Resource-Policy;
        proxy_hide_header Access-Control-Allow-Origin;
        add_header Cross-Origin-Resource-Policy "cross-origin" always;
        add_header Access-Control-Allow-Origin "*" always;
        add_header Cache-Control "public, max-age=604800" always;
        proxy_intercept_errors on;
        proxy_read_timeout 30s;
    }

    # Service worker + workbox runtime MUST never be HTTP-cached, otherwise the
    # browser keeps serving the old SW with stale precache lists pointing at
    # asset hashes that no longer exist (bad-precaching-response 404s).
    # These need the `^~` and `=` prefix to win over the asset regex below.
    location = /sw.js {
        try_files $uri =404;
        add_header Cache-Control "no-cache, no-store, must-revalidate" always;
        add_header Pragma "no-cache" always;
        add_header Expires "0" always;
    }
    location ^~ /workbox- {
        try_files $uri =404;
        add_header Cache-Control "no-cache, no-store, must-revalidate" always;
        add_header Pragma "no-cache" always;
        add_header Expires "0" always;
    }
    # index.html also revalidates so SW registration always sees fresh asset map.
    location = /index.html {
        try_files $uri =404;
        add_header Cache-Control "no-cache, no-store, must-revalidate" always;
    }
    location = / {
        try_files /index.html =404;
        add_header Cache-Control "no-cache, no-store, must-revalidate" always;
    }

    location ~* \.(?:css|js|mjs|map|json|wasm|mp3|flac|wav|ogg|png|jpg|jpeg|svg|webp|ico|gz|br|utf|ttf|woff2?)$ {
        try_files $uri =404;
        expires 30d;
        add_header Cache-Control "public, immutable";
    }

    location / {
        try_files $uri $uri/ /index.html;
    }
}
